#include "../Headers/Vehicle.h"

template<typename T>
Vehicle<T>::Vehicle(T Availability) {
    this->Availability = Availability;
}

template<typename T>
T Vehicle<T>::GetAvailability() const {
    return this->Availability;
}

template<typename T>
void Vehicle<T>::SetAvailability(T Availability) {
    this->Availability = Availability;
}

template<typename T>
std::ostream &operator << (std::ostream &out, Vehicle<T> &vehicle) {
    return vehicle.Print(out);
}

template<typename T>
std::ostream &Vehicle<T>::Print(std::ostream &out) {
    out << "Available: " << this->GetAvailability() << std::endl;
    return out;
}
